🌍 Project: Incident Management Dashboard

Create a dashboard that shows incident statistics and alerts admins about high-priority issues.

🎯 Complete Application Code:
// Incident Dashboard Application function IncidentDashboard() { this.init(); } IncidentDashboard.prototype = { init: function() { this.getCurrentUserInfo(); this.getIncidentStats(); this.checkHighPriority(); }, getCurrentUserInfo: function() { var currentUser = gs.getUser(); gs.info('=== USER INFO ==='); gs.info('Name: ' + currentUser.getName()); gs.info('Email: ' + currentUser.getEmail()); gs.info(''); }, getIncidentStats: function() { gs.info('=== INCIDENT STATISTICS ==='); var statuses = ['new', 'in_progress', 'on_hold', 'resolved']; var stats = {}; for (var i = 0; i < statuses.length; i++) { var gr = new GlideRecord('incident'); gr.addQuery('state', statuses[i]); gr.query(); stats[statuses[i]] = gr.getRowCount(); gs.info(statuses[i] + ': ' + gr.getRowCount()); } gs.info(''); }, checkHighPriority: function() { gs.info('=== HIGH PRIORITY INCIDENTS ==='); var gr = new GlideRecord('incident'); gr.addQuery('priority', '1'); gr.addQuery('state', '!=', 'resolved'); gr.orderByDesc('created_on'); gr.setLimit(5); gr.query(); if (gr.getRowCount() === 0) { gs.info('✅ No high-priority incidents!'); } else { while (gr.next()) { gs.info('🔴 ' + gr.getValue('number') + ' - ' + gr.getDisplayValue('assigned_to')); } } }, checkUserRole: function(roleName) { var gr = new GlideRecord('sys_user_role'); gr.addQuery('user', gs.getUser().getID()); gr.addQuery('role.name', roleName); gr.query(); return gr.getRowCount() > 0; } }; // Run the dashboard var dashboard = new IncidentDashboard();

📋 What This Code Does:

  • ✅ Gets current user information
  • ✅ Counts incidents by status
  • ✅ Finds high-priority incidents
  • ✅ Shows who incidents are assigned to
  • ✅ Checks user roles
