🌍 REAL: Safe Query Execution
try { var gr = new GlideRecord('incident'); gr.addQuery('number', 'INC0001'); gr.query(); if (gr.next()) { gs.info('Found: ' + gr.getValue('short_description')); } else { gs.info('Incident not found'); } } catch (e) { gs.info('Error querying incidents: ' + e); }

📖 Classes - Organize Code

var User = Class.create(); User.prototype = { initialize: function(name, email) { this.name = name; this.email = email; }, getInfo: function() { return this.name + ' (' + this.email + ')'; } }; var user = new User('Alice', 'alice@example.com'); gs.info(user.getInfo());
🌍 REAL: User Permission Class
var UserPermissions = Class.create(); UserPermissions.prototype = { initialize: function(userId) { this.userId = userId; }, hasRole: function(roleName) { var gr = new GlideRecord('sys_user_role'); gr.addQuery('user', this.userId); gr.addQuery('role.name', roleName); gr.query(); return gr.getRowCount() > 0; }, canEdit: function() { return this.hasRole('editor') || this.hasRole('admin'); } }; var perms = new UserPermissions(gs.getUser().getID()); if (perms.canEdit()) { gs.info('User can edit records'); }
