Day 8: GlideRecord

📖 What is GlideRecord?

GlideRecord queries ServiceNow database tables.

var gr = new GlideRecord('incident'); gr.query(); while (gr.next()) { gs.info(gr.getValue('number')); }
🌍 REAL: Get All High Priority Incidents
var gr = new GlideRecord('incident'); gr.addQuery('priority', '1'); // 1 = Critical gr.addQuery('state', '!=', '7'); // Not closed gr.orderByDesc('created_on'); gr.setLimit(10); gr.query(); while (gr.next()) { gs.info('Incident: ' + gr.getValue('number')); gs.info('Assigned: ' + gr.getDisplayValue('assigned_to')); }
🌍 Get Users by Role
var gr = new GlideRecord('sys_user'); gr.addQuery('active', 'true'); gr.query(); var adminCount = 0; while (gr.next()) { var userID = gr.getValue('sys_id'); var roleGr = new GlideRecord('sys_user_role'); roleGr.addQuery('user', userID); roleGr.addQuery('role.name', 'admin'); roleGr.query(); if (roleGr.getRowCount() > 0) { adminCount++; } } gs.info('Total Admins: ' + adminCount);
🌍 Fetch VIP Customers
var gr = new GlideRecord('customer'); gr.addQuery('vip', true); gr.orderBy('name'); gr.query(); gs.info('VIP Customers:'); while (gr.next()) { var name = gr.getValue('name'); var account = gr.getValue('account_number'); gs.info(' ' + name + ' (Account: ' + account + ')'); }
